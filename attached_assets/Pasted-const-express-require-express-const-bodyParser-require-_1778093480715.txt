const express = require('express');
const bodyParser = require('body-parser');
const axios = require('axios');
const app = express();
const port = 3000;

// Gunakan body-parser untuk mem-parsing body request
app.use(bodyParser.json());

// Nomor vendor berdasarkan jenis layanan yang dipilih
const vendorNumbers = {
  "Trucking": "whatsapp:+1234567890", // Vendor Trucking
  "Handling": "whatsapp:+0987654321", // Vendor Handling
  "Custom Clearance": "whatsapp:+1122334455" // Vendor Custom Clearance
};

// Endpoint untuk menerima permintaan dari Fonnte
app.post('/order', async (req, res) => {
  const orderData = req.body;

  // Mendapatkan jenis layanan dari form
  const serviceType = orderData.fields.find(field => field.label === "Jenis Layanan").value; // Mendapatkan jenis layanan yang dipilih oleh CS

  // Pilih nomor vendor berdasarkan jenis layanan
  const vendorPhone = vendorNumbers[serviceType];

  if (!vendorPhone) {
    return res.status(400).send('Jenis layanan tidak valid');
  }

  // Format pesan yang akan dikirimkan ke vendor dan admin
  const message = `Pesanan Anda Diterima:\n\n` +
    `No. Order: ${orderData.fields.find(field => field.label === "No. Order").value}\n` +
    `Status: Menunggu Konfirmasi\n` +
    `Jenis: ${orderData.fields.find(field => field.label === "Jenis").value}\n` +
    `Rute: ${orderData.fields.find(field => field.label === "Rute").value}\n` +
    `Kategori Barang: ${orderData.fields.find(field => field.label === "Kategori Barang").value}\n` +
    `Layanan: ${serviceType}\n` +
    `Total Est: Rp ${orderData.fields.find(field => field.label === "Total Est").value}\n\n` +
    `Tim kami akan segera menghubungi Anda untuk konfirmasi lebih lanjut.`

  try {
    // Kirim pesan ke nomor vendor yang dipilih
    await axios.post('https://api.whatsapp.com/send', {
      phone: vendorPhone,
      message: message
    });

    // Kirim pesan ke admin (nomor admin sudah statis)
    await axios.post('https://api.whatsapp.com/send', {
      phone: "whatsapp:+0987654321",  // Nomor Admin
      message: message
    });

    res.status(200).send('Order diteruskan ke vendor dan admin');
  } catch (error) {
    console.error('Error forwarding order:', error);
    res.status(500).send('Terjadi kesalahan dalam mengirim data ke vendor dan admin');
  }
});

// Jalankan server di port 3000
app.listen(port, () => {
  console.log(`Server berjalan di http://localhost:${port}`);
});
